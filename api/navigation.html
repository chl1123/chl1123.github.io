    @classmethod
@call_service("DSPChassis")
def getChassisStop(cls) -> bool:
    """底盘是否停止（仅通过walk电机判断）

    Returns:
        bool: 如果行走电机停止则为True
    """
    is_stop = cls.client().call_service("DSPChassis", "getChassisStop")
    if is_stop:
        is_stop = True
    else:
        if cls.not_stop_counts >= 1:
            is_stop = False
            cls.not_stop_counts = 0
        else:
            # 非停止状态计数
            cls.not_stop_counts += 1
    return is_stop
